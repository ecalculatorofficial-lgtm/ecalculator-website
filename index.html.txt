        });

        function initializeApp() {
            generateQuickTargets();
            setupEventListeners();
            applyTheme();
            showPage('calculator');
            setupDragFunctionality();
        }

        function setupEventListeners() {
            const incomeInput = document.getElementById('monthly-income');
            const expensesInput = document.getElementById('monthly-expenses');
            const customTargetInput = document.getElementById('custom-target');
           
            incomeInput?.addEventListener('input', calculateSavings);
            expensesInput?.addEventListener('input', calculateSavings);
            customTargetInput?.addEventListener('input', handleCustomTarget);

            // Contact form
            const contactForm = document.getElementById('contact-form');
            contactForm?.addEventListener('submit', handleContactForm);
        }

        // Enhanced Cursor-Following Drag Functionality for Back Button
        function setupDragFunctionality() {
            const backBtn = document.getElementById('back-btn');
            if (!backBtn) return;

            // Mouse events
            document.addEventListener('mousemove', handleDragMove);
            document.addEventListener('mouseup', handleDragEnd);

            // Touch events for mobile
            document.addEventListener('touchmove', handleDragMove, { passive: false });
            document.addEventListener('touchend', handleDragEnd);
        }

        function startDrag(event) {
            event.preventDefault();
            isDragging = true;
           
            const backBtn = document.getElementById('back-btn');
            backBtn.classList.add('dragging');
           
            const clientX = event.type === 'mousedown' ? event.clientX : event.touches[0].clientX;
            const clientY = event.type === 'mousedown' ? event.clientY : event.touches[0].clientY;
           
            const rect = backBtn.getBoundingClientRect();
            dragOffset.x = clientX - rect.left;
            dragOffset.y = clientY - rect.top;
        }

        function handleDragMove(event) {
            if (!isDragging) return;
           
            event.preventDefault();
           
            if (animationFrame) {
                cancelAnimationFrame(animationFrame);
            }
           
            animationFrame = requestAnimationFrame(() => {
                const backBtn = document.getElementById('back-btn');
               
                const clientX = event.type === 'mousemove' ? event.clientX : event.touches[0].clientX;
                const clientY = event.type === 'mousemove' ? event.clientY : event.touches[0].clientY;
               
                let newX = clientX - dragOffset.x;
                let newY = clientY - dragOffset.y;
               
                // Constrain to viewport with smooth boundaries
                const maxX = window.innerWidth - backBtn.offsetWidth;
                const maxY = window.innerHeight - backBtn.offsetHeight;
               
                newX = Math.max(0, Math.min(newX, maxX));
                newY = Math.max(0, Math.min(newY, maxY));
               
                backBtn.style.left = newX + 'px';
                backBtn.style.top = newY + 'px';
            });
        }

        function handleDragEnd(event) {
            if (!isDragging) return;
           
            isDragging = false;
            const backBtn = document.getElementById('back-btn');
            backBtn.classList.remove('dragging');
           
            if (animationFrame) {
                cancelAnimationFrame(animationFrame);
            }
           
            // Save position to localStorage
            const rect = backBtn.getBoundingClientRect();
            localStorage.setItem('backBtnPosition', JSON.stringify({
                x: rect.left,
                y: rect.top
            }));
        }

        function generateQuickTargets() {
            const container = document.getElementById('quick-targets');
            if (!container) return;

            container.innerHTML = quickTargets.map(amount => `
                <button onclick="selectTarget(${amount})"
                        class="target-btn px-4 py-3 rounded-lg font-medium hover:scale-105 transition-all">
                    ${formatCurrency(amount)}
                </button>
            `).join('');
        }

        function showPage(pageId) {
            // Store previous page for back functionality
            if (currentPage !== pageId) {
                previousPage = currentPage;
            }
           
            // Hide all pages
            document.querySelectorAll('.page').forEach(page => {
                page.classList.add('hidden');
            });
           
            // Show selected page
            const targetPage = document.getElementById(`${pageId}-page`);
            if (targetPage) {
                targetPage.classList.remove('hidden');
                currentPage = pageId;
               
                // Show/hide back button
                const backBtn = document.getElementById('back-btn');
                if (pageId === 'calculator') {
                    backBtn.classList.add('hidden');
                    backBtn.style.display = 'none';
                } else {
                    backBtn.classList.remove('hidden');
                    backBtn.style.display = 'flex';
                   
                    // Restore saved position
                    const savedPosition = localStorage.getItem('backBtnPosition');
                    if (savedPosition) {
                        const position = JSON.parse(savedPosition);
                        backBtn.style.left = position.x + 'px';
                        backBtn.style.top = position.y + 'px';
                    }
                }
               
                // Add slide-in animation
                targetPage.querySelector('.slide-in')?.classList.add('slide-in');
            }

            // Close mobile menu
            const mobileMenu = document.getElementById('mobile-menu');
            if (mobileMenu) {
                mobileMenu.classList.add('hidden');
                document.getElementById('mobile-menu-icon').className = 'fas fa-bars';
            }
        }

        function goBack() {
            if (previousPage) {
                showPage(previousPage);
            } else {
                showPage('calculator');
            }
        }

        function toggleTheme() {
            isDarkMode = !isDarkMode;
            localStorage.setItem('theme', isDarkMode ? 'dark' : 'light');
            applyTheme();
        }

        function applyTheme() {
            const body = document.body;
            const themeIcon = document.getElementById('theme-icon');
           
            if (isDarkMode) {
                body.classList.add('dark');
                themeIcon.className = 'fas fa-sun';
            } else {
                body.classList.remove('dark');
                themeIcon.className = 'fas fa-moon';
            }
        }

        function toggleMobileMenu() {
            const mobileMenu = document.getElementById('mobile-menu');
            const icon = document.getElementById('mobile-menu-icon');
           
            if (mobileMenu.classList.contains('hidden')) {
                mobileMenu.classList.remove('hidden');
                icon.className = 'fas fa-times';
            } else {
                mobileMenu.classList.add('hidden');
                icon.className = 'fas fa-bars';
            }
        }

        function calculateSavings() {
            const income = parseFloat(document.getElementById('monthly-income')?.value) || 0;
            const expenses = parseFloat(document.getElementById('monthly-expenses')?.value) || 0;
            const savings = Math.max(0, income - expenses);
           
            const savingsDisplay = document.getElementById('monthly-savings');
            if (savingsDisplay) {
                savingsDisplay.textContent = '$' + formatNumberPrecise(savings);
            }

            // Update progress bar
            const progressBar = document.getElementById('savings-progress');
            if (progressBar && income > 0) {
                const percentage = Math.min(100, (savings / income) * 100);
                progressBar.style.width = `${percentage}%`;
            }

            // Auto-navigate to results if target is selected and all fields filled - ADDED
            if (selectedTarget > 0 && income > 0 && expenses >= 0) {
                calculateResults();
            }
        }

        function selectTarget(amount) {
            selectedTarget = amount;
           
            // Update custom input
            const customInput = document.getElementById('custom-target');
            if (customInput) {
                customInput.value = formatNumberPrecise(amount);
            }
           
            // Update button styles
            document.querySelectorAll('.target-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
           
            // Auto-navigate to results if income and expenses are filled - ADDED
            const income = parseFloat(document.getElementById('monthly-income')?.value) || 0;
            const expenses = parseFloat(document.getElementById('monthly-expenses')?.value) || 0;
           
            if (income > 0 && expenses >= 0) {
                calculateResults();
            }
        }

        function handleCustomTarget(event) {
            const value = event.target.value.replace(/[^\d.]/g, '');
            event.target.value = value;
           
            const amount = parseFloat(value) || 0;
            selectedTarget = amount;
           
            // Remove active class from quick targets
            document.querySelectorAll('.target-btn').forEach(btn => {
                btn.classList.remove('active');
            });
           
            // Auto-navigate to results if amount is valid and other fields are filled - ADDED
            if (amount > 0) {
                const income = parseFloat(document.getElementById('monthly-income')?.value) || 0;
                const expenses = parseFloat(document.getElementById('monthly-expenses')?.value) || 0;
               
                if (income > 0 && expenses >= 0) {
                    calculateResults();
                }
            }
        }

        function calculateResults() {
            const income = parseFloat(document.getElementById('monthly-income')?.value) || 0;
            const expenses = parseFloat(document.getElementById('monthly-expenses')?.value) || 0;
            const savings = Math.max(0, income - expenses);
           
            if (selectedTarget <= 0 || savings <= 0) {
                document.getElementById('results-section')?.classList.add('hidden');
                return;
            }
           
            // Enhanced calculation with precision
            const exactMonthsToGoal = selectedTarget / savings;
            const monthsToGoal = Math.ceil(exactMonthsToGoal);
            const years = Math.floor(monthsToGoal / 12);
            const months = monthsToGoal % 12;
           
            // Update result displays with precise formatting
            document.getElementById('result-target').textContent = formatCurrency(selectedTarget);
            document.getElementById('result-time').textContent =
                years > 0 ? `${years}y ${months}m` : `${monthsToGoal} months`;
            document.getElementById('result-savings').textContent = '$' + formatNumberPrecise(savings);
           
            // Show results section
            document.getElementById('results-section')?.classList.remove('hidden');
           
            // Auto-scroll to results section - ADDED
            const resultsSection = document.getElementById('results-section');
            if (resultsSection) {
                resultsSection.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
           
            // Generate chart with enhanced accuracy
            generateChart(savings, selectedTarget, exactMonthsToGoal);
        }

        function generateChart(monthlySavings, targetAmount, totalMonths) {
            const ctx = document.getElementById('savingsChart');
            if (!ctx) return;
           
            // Destroy existing chart
            if (savingsChart) {
                savingsChart.destroy();
            }
           
            // Generate data points with higher precision
            const dataPoints = [];
            const labels = [];
            const maxPoints = 60;
            const stepSize = Math.max(1, totalMonths / maxPoints);
           
            for (let month = 0; month <= totalMonths; month += stepSize) {
                const currentAmount = month * monthlySavings;
                dataPoints.push(Math.min(currentAmount, targetAmount));
               
                if (month === 0) {
                    labels.push('Start');
                } else if (month >= 12) {
                    const years = Math.floor(month / 12);
                    const remainingMonths = Math.round(month % 12);
                    labels.push(remainingMonths > 0 ? `${years}y ${remainingMonths}m` : `${years}y`);
                } else {
                    labels.push(`${Math.round(month)}m`);
                }
            }
           
            // Ensure final point is included
            if (dataPoints[dataPoints.length - 1] < targetAmount) {
                dataPoints.push(targetAmount);
                const finalYears = Math.floor(totalMonths / 12);
                const finalMonths = Math.round(totalMonths % 12);
                labels.push(finalYears > 0 ? `${finalYears}y ${finalMonths}m` : `${Math.round(totalMonths)}m`);
            }
           
            // Enhanced color settings for better dark mode visibility - FIXED
            const primaryTextColor = isDarkMode ? '#F5DEB3' : '#5D4E37';
            const secondaryTextColor = isDarkMode ? '#DEB887' : '#8B4513';
            const gridColor = isDarkMode ? 'rgba(245, 222, 179, 0.25)' : 'rgba(139, 69, 19, 0.1)';
           
            savingsChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [{
                        label: 'Savings Growth',
                        data: dataPoints,
                        borderColor: '#8B4513',
                        backgroundColor: 'rgba(139, 69, 19, 0.1)',
                        fill: true,
                        tension: 0.4,
                        pointBackgroundColor: '#CD853F',
                        pointBorderColor: '#CD853F',
                        pointRadius: 4
                    }, {
                        label: 'Target Amount',
                        data: new Array(labels.length).fill(targetAmount),
                        borderColor: '#CD853F',
                        backgroundColor: 'transparent',
                        borderDash: [5, 5],
                        pointRadius: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            labels: {
                                color: primaryTextColor,
                                font: {
                                    weight: 'bold',
                                    size: 14
                                }
                            }
                        }
                    },
                    scales: {
                        x: {
                            ticks: {
                                color: secondaryTextColor,
                                font: {
                                    weight: '600'
                                }
                            },
                            grid: {
                                color: gridColor
                            }
                        },
                        y: {
                            ticks: {
                                color: secondaryTextColor,
                                font: {
                                    weight: '600'
                                },
                                callback: function(value) {
                                    return formatCurrency(value);
                                }
                            },
                            grid: {
                                color: gridColor
                            }
                        }
                    }
                }
            });
        }

        function handleContactForm(event) {
            event.preventDefault();
           
            // Show success message
            const form = event.target;
            const originalContent = form.innerHTML;
           
            form.innerHTML = `
                <div class="text-center py-12">
                    <div class="w-16 h-16 gradient-btn rounded-full flex items-center justify-center mx-auto mb-4 pulse-glow">
                        <i class="fas fa-check text-white text-2xl"></i>
                    </div>
                    <h3 class="text-lg font-semibold text-primary mb-2">Message Sent!</h3>
                    <p class="text-secondary">Thank you for your message! We'll get back to you soon.</p>
                </div>
            `;
           
            // Reset form after 3 seconds
            setTimeout(() => {
                form.innerHTML = originalContent;
                setupEventListeners();
            }, 3000);
        }

        // Enhanced utility functions with improved precision
        function formatNumberPrecise(num) {
            return new Intl.NumberFormat('en-US', {
                minimumFractionDigits: 0,
                maximumFractionDigits: 2
            }).format(num);
        }

        function formatCurrency(amount) {
            if (amount >= 1000000000000000) {
                return `$${(amount / 1000000000000000).toFixed(amount % 1000000000000000 === 0 ? 0 : 1)}Q`;
            } else if (amount >= 1000000000000) {
                return `$${(amount / 1000000000000).toFixed(amount % 1000000000000 === 0 ? 0 : 1)}T`;
            } else if (amount >= 1000000000) {
                return `$${(amount / 1000000000).toFixed(amount % 1000000000 === 0 ? 0 : 1)}B`;
            } else if (amount >= 1000000) {
                return `$${(amount / 1000000).toFixed(amount % 1000000 === 0 ? 0 : 1)}M`;
            } else if (amount >= 1000) {
                return `$${(amount / 1000).toFixed(amount % 1000 === 0 ? 0 : 0)}K`;
            } else {
                return `$${formatNumberPrecise(amount)}`;
            }
        }
